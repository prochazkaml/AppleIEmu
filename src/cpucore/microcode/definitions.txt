__CARRY_FLAG 0x01
__NOT_CARRY_FLAG 0xFE
__ZERO_FLAG 0x02
__NOT_ZERO_FLAG 0xFD
__INTERRUPT_FLAG 0x04
__NOT_INTERRUPT_FLAG 0xFB
__DECIMAL_FLAG 0x08
__NOT_DECIMAL_FLAG 0xF7
__OVERFLOW_FLAG 0x40
__NOT_OVERFLOW_FLAG 0xBF
__NEGATIVE_FLAG 0x80
__NOT_NEGATIVE_FLAG 0x7F

SET_CARRY_FLAG p = bitor(p, __CARRY_FLAG)
CLR_CARRY_FLAG p = bitand(p, __NOT_CARRY_FLAG)
SET_ZERO_FLAG p = bitor(p, __ZERO_FLAG)
CLR_ZERO_FLAG p = bitand(p, __NOT_ZERO_FLAG)
SET_INTERRUPT_FLAG p = bitor(p, __INTERRUPT_FLAG)
CLR_INTERRUPT_FLAG p = bitand(p, __NOT_INTERRUPT_FLAG)
SET_DECIMAL_FLAG p = bitor(p, __DECIMAL_FLAG)
CLR_DECIMAL_FLAG p = bitand(p, __NOT_DECIMAL_FLAG)
SET_OVERFLOW_FLAG p = bitor(p, __OVERFLOW_FLAG)
CLR_OVERFLOW_FLAG p = bitand(p, __NOT_OVERFLOW_FLAG)
SET_NEGATIVE_FLAG p = bitor(p, __NEGATIVE_FLAG)
CLR_NEGATIVE_FLAG p = bitand(p, __NOT_NEGATIVE_FLAG)
